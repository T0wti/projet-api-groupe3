UserService :
-- Table des informations de compte
CREATE TABLE users (
    id UUID PRIMARY KEY, -- Même UUID que l'auth
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    bio TEXT,
    avatar_url VARCHAR(255),
    language_preference VARCHAR(10) DEFAULT 'fr', -- Interface multi-langues (Fx22)
    theme_preference VARCHAR(20) DEFAULT 'light' -- Thème personnalisé (Fx23)
);
